## 코드(C언어)
'''c
// 피보나치 함수(1003번)
#include <stdio.h>
int main()
{
   int n;
   scanf_s("%d", &n);
   int i;
   int arr[100][2], result[100][2];
   arr[0][0] = 1;
   arr[0][1] = 0;
   arr[1][0] = 0;
   arr[1][1] = 1;
   for (i = 2; i <= 40; i++)
   {
      arr[i][0] = arr[i - 1][0] + arr[i - 2][0];
      arr[i][1] = arr[i - 1][1] + arr[i - 2][1];
   }

   int fib, max = 0;
   for (i = 1; i <= n; i++)
   {
      scanf_s("%d", &fib);
      printf("%d %d\n", arr[fib][0], arr[fib][1]);
   }

   return 0;
}
'''
