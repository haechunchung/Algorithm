## 코드(Python)
```Python
# 토마토(7576번)
import sys
input = sys.stdin.readline
sys.setrecursionlimit(10**9)

m, n = input().split()
m = int(m)
n = int(n)

box = []
for i in range(n):
    box.append(list(map(int, input().split())))

bad_tomato = 0
good_tomato = []

for i in range(n):
    bad_tomato += box[i].count(0)
    for j in range(m):
        if box[i][j] == 1:
            good_tomato.append([i, j])

if bad_tomato == 0:
    print(0)
else:
    time = 0
    shift = [[1, 0], [0, 1], [-1, 0], [0, -1]]

    while 1:
        if bad_tomato == 0:
            print(time)
            sys.exit(0)

        if not good_tomato:
            print(-1)
            sys.exit(0)

        k = len(good_tomato)
        for r in range(k):
            i = good_tomato[r][0]
            j = good_tomato[r][1]
            for x, y in shift:
                if 0 <= i + x and i + x < n and 0 <= j + y and j + y < m and box[i + x][j + y] == 0:
                    box[i + x][j + y] = 1
                    good_tomato.append([i + x, j + y])
                    bad_tomato -= 1

        del good_tomato[:k]
        time += 1
```
