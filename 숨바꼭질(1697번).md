## 코드(Python)
```Python
# 숨바꼭질(1697번)
import sys
from collections import deque
input = sys.stdin.readline
n, k = input().split()
n = int(n)
k = int(k)
check = {n}
visit = deque([n])
count = 0
if n >= k:
    print(n - k)
else:
    while 1:
        count += 1
        t = len(visit)
        for j in range(t):
            i = visit.pop()
            move = [i - 1, i + 1, i * 2]
            if k in move:
                print(count)
                sys.exit(0)
            for a in move:
                if a >= 0 and a <= 100000 and a not in check:
                    check.add(a)
                    visit.appendleft(a)
```
