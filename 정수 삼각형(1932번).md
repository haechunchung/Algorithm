## 코드(C언어)

// 정수 삼각형(1932번)
#include <stdio.h>
int main()
{
   int height;
   scanf_s("%d", &height);
   int i, j;
   int num[501][501] = { 0 };
   int max = 0;
   for (i = 1; i <= height; i++)
   {
      for (j = 1; j <= i; j++)
         scanf_s("%d", &num[i][j]);
   }
   if (height == 1)
      max = num[1][1];
   else
   {
      for (i = 2; i <= height; i++)
      {
         for (j = 1; j <= i; j++)
         {
            if (j == 1)
               num[i][j] = num[i][j] + num[i - 1][j];
            else if (j == i)
               num[i][j] = num[i][j] + num[i - 1][j - 1];
            else if (num[i - 1][j] >= num[i - 1][j - 1])
               num[i][j] = num[i][j] + num[i - 1][j];
            else num[i][j] = num[i][j] + num[i - 1][j - 1];

            if (i == height)
            {
               if (max < num[i][j])
                  max = num[i][j];
            }
         }
      }
   }
   printf("%d\n", max);
   return 0;
}
