## 코드(Python)
```Python
# 컵라면(1781번)
import sys
input = sys.stdin.readline
n = int(input())
problem = {}
for i in range(n):
    a, b = input().split()
    if int(a) in problem.keys():
        problem[int(a)].append(int(b))
    else:
        problem[int(a)] = [int(b)]
problem = sorted(problem.items())
result = []
for deadline, k in problem:
    result += k
    while len(result) > deadline:
        result.remove(min(result))
print(sum(result))
```
